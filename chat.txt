New Function Additions:

python
from google import genai
from google.genai import types
import json
import os
# You would set this in your terminal before running: export GEMINI_API_KEY="your_key"
client = genai.Client()
def extract_item_details_from_text(user_text):
    prompt = f"""
    You are an assistant for a campus Lost and Found portal. Extract the following details from the user's report:
    - Item Name (e.g., 'MacBook Pro', 'Keys', 'Wallet')
    - Description (color, brand, any identifying features)
    - Location (where it was lost or found)
    - Date (if mentioned, otherwise estimate based on 'yesterday', etc., or leave blank)
    
    User Report: "{user_text}"
    
    Return the result ONLY as a valid JSON object with keys: "name", "description", "location", "date".
    """
    
    response = client.models.generate_content(
        model='gemini-2.5-flash',
        contents=prompt,
        config=types.GenerateContentConfig(
            response_mime_type="application/json",
        ),
    )
    
    # Parse the JSON response
    return json.loads(response.text)
New Route Addition: We'll create a new backend route called /ai-parse-report that the frontend can talk to.

@app.route('/ai-parse-report', methods=['POST'])
def ai_parse_report():
    data = request.get_json()
    user_text = data.get('text')
    
    try:
        extracted_data = extract_item_details_from_text(user_text)
        return {"success": True, "data": extracted_data}
    except Exception as e:
        return {"success": False, "error": str(e)}
